R10 = 2, R11 = 3, R12 = 4
MEM[2] = 10.23, MEM[15] = 3.14

ld F13, 0(R10)  		//F13 = MEM[R10 + 0] = MEM[2] = 10.23
ld F14, 10(R11)  		//F14 = MEM[R11 + 3] = MEM[15] = 3.14
add.d F14, F14, F13     //F14 = F14 + F13
addi R10, R10, 10       //R10 = R10 + 10
add R10, R10, R12       //R10 = R10 + R12
sub R11, R12, R11       //R11 = R12 - R11
sd F13, 0(R11)          //MEM[1] = F13 = 10.23
sd F14, 0(R10)          //MEM[16] = F14 = 3.14
